python -c "import torch; print('CUDA Available:', torch.cuda.is_available())"


## 统一转成 24kHz、单声道 wav
mkdir -p assets/BZNSYP_24k/Wave
for f in assets/BZNSYP/Wave/*.wav; do
  ffmpeg -y -i "$f" -ac 1 -ar 24000 "assets/BZNSYP_24k/Wave/$(basename "$f")"
done


## 如何选固定 ref.wav？
从 Wave/ 里挑一条同说话人、干净、6–15 秒、语气平稳、少静音、少异常文本的 wav，然后转成 24kHz mono。官方建议整个数据集共用同一个 ref.wav。
选中的是 assets/BZNSYP_24k/Wave/001760.wav --> assets/BZNSYP_24k/ref.wav

## 这个已经被删除了,在 assets/BZNSYP_24k/ft_data/train_raw.jsonl 中原本是6626行处
{"audio": "assets/BZNSYP_24k/Wave/001760.wav", "text": "粘苍蝇的纸放一天会粘满苍蝇，二货朋友说为什么都那么傻，没看到同伴都死了吗？", "ref_audio": "assets/BZNSYP_24k/ref.wav"}


## 生成训练数据jsonl文件
python finetuning/script/raw2jsonl.py \
  --audio_dir assets/BZNSYP_24k/Wave \
  --prosody_dir assets/BZNSYP/ProsodyLabeling \
  --ref_audio assets/BZNSYP_24k/ref.wav \
  --out_dir assets/BZNSYP_24k/ft_data \
  --val_size 100 \
  --test_size 100 \
  --seed 42 \
  --relative_to ./

## 使用语音分词器Qwen3-TTS-Tokenizer-12Hz
python finetuning/prepare_data.py \
  --device cuda:0 \
  --tokenizer_model_path ./Qwen3-TTS-Tokenizer-12Hz \
  --input_jsonl assets/BZNSYP_24k/ft_data/train_raw.jsonl \
  --output_jsonl assets/BZNSYP_24k/codec/train_with_codes.jsonl


## SFT全量训练
python finetuning/sft_12hz_with_metrics.py \
  --init_model_path ./Qwen3-TTS-12Hz-1.7B-Base \
  --output_model_path finetuning/exp/output_bznsyp_1p7b_sft-3.21 \
  --train_jsonl assets/BZNSYP_24k/codec/train_with_codes.jsonl \
  --batch_size 1 \
  --lr 1e-5 \
  --num_epochs 2 \
  --speaker_name bznsyp_female \
  --save_plots_every_epoch \
  --log_with tensorboard,wandb
